String::replace()

Replace all occurrences of before with after in this string.

Examples 
"abc xabc".replace("abc", "def") //-> "def xdef"
Source Code 
public method replace(this: String, before: String, after: String): String {
  let parts: List<String> = []

  let s = this
  while True {
    match s.index_of(before) {
      Some(i) => {
        parts = parts.append(s.substring(0, i))
        parts = parts.append(after)
        s = s.substring(i + before.len(), s.len())
      }
      None => {
        parts = parts.append(s)
        break
      }
    }
  }

  "".join(parts)
}
